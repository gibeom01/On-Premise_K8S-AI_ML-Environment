# Swap 비활성화 -> 메모리 스와핑이 발생하면 파드 스케줄링 성능이 저하되므로 Swap 완전히 꺼야 함.

# 현재 적용된 Swap 메모리 끄기
1. sudo swapoff -a 

# 재부팅 후에도 Swap 켜지지 않도록 fstab 파일 수정 (swap이 들어간 라인 주석 처리)
2. sudo sed -i '/ swap / s/^\(.*\)$/#\1/g' /etc/fstab

---

# 커널 파라미터 적용 -> 네트워크 패킷 포워딩 및 브릿지 트래픽 iptables가 볼 수 있도록 커널 모듈 로드.

1. cat <<EOF | sudo tee /etc/modules-load.d/k8s.conf
overlay
br_netfilter
EOF
2. sudo modprobe overlay
3. sudo modprobe br_netfilter

# 커널 파라미터 영구 설정
4. cat <<EOF | sudo tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward                 = 1
EOF
5. sudo sysctl --system

---

# 컨테이너 런타임 설치 (containerd) -> 파드(컨테이너) 실제 띄워줄 엔진 설치.

# 필수 패키지 설치
1. sudo apt update
2. sudo apt install -y apt-transport-https ca-certificates curl gnupg lsb-release

# Docker 공식 GPG 키 추가 (containerd는 Docker 패키지 저장소에서 제공)
3. curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

# 저장소 추가
4. echo "deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

# containerd 설치
5. sudo apt update
6. sudo apt install -y containerd.io

# containerd 기본 설정 파일 생성 및 SystemdCgroup 활성화
7. sudo mkdir -p /etc/containerd
8. containerd config default | sudo tee /etc/containerd/config.toml
9. sudo sed -i 's/SystemdCgroup = false/SystemdCgroup = true/g' /etc/containerd/config.toml

# 데몬 재시작 및 자동 실행 등록
10. sudo systemctl restart containerd
11. sudo systemctl enable containerd

---

# K8s 코어 패키지 설치 (kubeadm, kubelet) -> 클러스터 구축하고 노드 관리할 쿠버네티스 패키지 설치.

# K8s 패키지 저장소용 공개 서명 키 다운로드 (v1.29 기준)
1. curl -fsSL https://pkgs.k8s.io/core:/stable:/v1.29/deb/Release.key | sudo gpg --dearmor -o /etc/apt/keyrings/kubernetes-apt-keyring.gpg

# K8s apt 리포지토리 추가
2. echo 'deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] https://pkgs.k8s.io/core:/stable:/v1.29/deb/ /' | sudo tee /etc/apt/sources.list.d/kubernetes.list

# 패키지 설치
3. sudo apt update
4. sudo apt install -y kubelet kubeadm kubectl

# apt upgrade 시 버전이 의도치 않게 올라가서 클러스터가 망가지는 것을 방지(Hold)
5. sudo apt-mark hold kubelet kubeadm kubectl

업데이트 문제 시 

# 락을 쥐고 있는 1592 프로세스 강제 종료 (숫자는 에러창에 뜬 PID에 맞게 입력)
1. sudo kill -9 1592

# 찌꺼기로 남은 잠금(Lock) 파일들 강제 삭제
2. sudo rm /var/lib/apt/lists/lock
3. sudo rm /var/cache/apt/archives/lock
4. sudo rm /var/lib/dpkg/lock*

# 꼬여있을 수 있는 패키지 설정 초기화
5. sudo dpkg --configure -a

# 다시 업데이트 시도
sudo apt update
